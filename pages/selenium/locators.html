Xpath 
                                                
                    // Absolute XPath - Uses a full path from the root element to the desired element.
                    By.xpath("/html/body/div[1]/div[2]/div[3]/button");
                    
                    // Relative XPath - Uses a shorter path starting from the current element or the current context.
                    By.xpath("//button[@id='submit']");
                    
                    // Basic XPath - Select an element by its tag name.
                    By.xpath("//tagName");
                    
                    // XPath with Attributes - Select elements based on attributes.
                    By.xpath("//input[@id='username']");
                    By.xpath("//button[@class='btn-primary']");
                    
                    // XPath with Multiple Attributes - Select elements based on multiple attributes.
                    By.xpath("//input[@id='username' and @name='user']");
                    
                    // XPath with Contains Function - Select elements containing a specific attribute value.
                    By.xpath("//input[contains(@name, 'user')]");
                    
                    // XPath with Starts-With Function - Select elements where an attribute value starts with a specific substring.
                    By.xpath("//input[starts-with(@name, 'user')]");
                    
                    // XPath with Text - Select elements based on their text content.
                    By.xpath("//a[text()='Sign In']");
                    
                    // XPath with Index - Select the nth element in a set of matching elements.
                    By.xpath("(//input[@type='text'])[2]");
                    
                    // XPath Axes
                    
                    // Ancestor Axis - Selects all ancestors (parent, grandparent, etc.) of the current node.
                    By.xpath("//div[@class='child']//ancestor::div");
                    
                    // Ancestor-Or-Self Axis - Selects all ancestors (parent, grandparent, etc.) and the current node.
                    By.xpath("//div[@class='child']//ancestor-or-self::div");
                    
                    // Attribute Axis - Selects all attributes of the current node.
                    By.xpath("//input[@id='username']/@*");
                    
                    // Child Axis - Selects all children of the current node.
                    By.xpath("//div[@class='parent']//child::input");
                    
                    // Descendant Axis - Selects all descendants (children, grandchildren, etc.) of the current node.
                    By.xpath("//div[@class='ancestor']//descendant::input");
                    
                    // Descendant-Or-Self Axis - Selects all descendants and the current node itself.
                    By.xpath("//div[@class='ancestor']//descendant-or-self::div");
                    
                    // Following Axis - Selects everything in the document after the closing tag of the current node.
                    By.xpath("//h2[@class='title']//following::div");
                    
                    // Following-Sibling Axis - Selects all siblings after the current node.
                    By.xpath("//div[@class='current']//following-sibling::div");
                    
                    // Parent Axis - Selects the parent of the current node.
                    By.xpath("//div[@class='child']//parent::div");
                    
                    // Preceding Axis - Selects all nodes before the current node in the document.
                    By.xpath("//h2[@class='title']//preceding::div");
                    
                    // Preceding-Sibling Axis - Selects all siblings before the current node.
                    By.xpath("//div[@class='current']//preceding-sibling::div");
                    
                    // Self Axis - Selects the current node.
                    By.xpath("//div[@class='current']//self::div");
                    
                    // XPath Operators
                    
                    // And Operator - Combines two conditions and returns true if both conditions are true.
                    By.xpath("//input[@type='text' and @name='username']");
                    
                    // Or Operator - Combines two conditions and returns true if at least one condition is true.
                    By.xpath("//input[@type='text' or @type='password']");
                    
                    // Not Operator - Negates the condition.
                    By.xpath("//input[not(@type='hidden')]");
                    
                    // Relational Operators
                    
                    // Equals Operator - Checks if the value of an attribute equals a specified value.
                    By.xpath("//input[@name='username' and @value='admin']");
                    
                    // Not Equals Operator - Checks if the value of an attribute does not equal a specified value.
                    By.xpath("//input[@name='username' and @value!='admin']");
                    
                    // Less Than Operator - Checks if the value of an attribute is less than a specified value.
                    By.xpath("//input[@value < 100]");
                    
                    // Greater Than Operator - Checks if the value of an attribute is greater than a specified value.
                    By.xpath("//input[@value > 100]");
                    
                    // Less Than or Equal To Operator - Checks if the value of an attribute is less than or equal to a specified value.
                    By.xpath("//input[@value <= 100]");
                    
                    // Greater Than or Equal To Operator - Checks if the value of an attribute is greater than or equal to a specified value.
                    By.xpath("//input[@value >= 100]");
                    
                    // Arithmetic Operators
                    
                    // Addition Operator - Adds two numeric values.
                    By.xpath("(//input[@type='text'])[position() + 1]");
                    
                    // Subtraction Operator - Subtracts one numeric value from another.
                    By.xpath("(//input[@type='text'])[position() - 1]");
                    
                    // Multiplication Operator - Multiplies two numeric values.
                    By.xpath("//input[@value * 2 = 200]");
                    
                    // Division Operator - Divides one numeric value by another.
                    By.xpath("//input[@value div 2 = 50]");
                    
                    // Modulus Operator - Returns the remainder of a division.
                    By.xpath("//input[@value mod 2 = 1]");
                    
                    // Union Operator - Combines the results of two XPath expressions.
                    By.xpath("//input[@type='text'] | //input[@type='password']");
                    
                    // XPath Functions
                    
                    // text() - Returns the text content of an element.
                    By.xpath("//a[text()='Sign In']");
                    
                    // contains() - Checks if a string contains a specified substring.
                    By.xpath("//input[contains(@name, 'user')]");
                    
                    // starts-with() - Checks if a string starts with a specified substring.
                    By.xpath("//input[starts-with(@name, 'user')]");
                    
                    // substring() - Returns a substring from a string.
                    By.xpath("//input[substring(@name, 1, 4) = 'user']");
                    
                    // substring-before() - Returns the substring before a specified delimiter.
                    By.xpath("//input[substring-before(@name, '_') = 'user']");
                    
                    // substring-after() - Returns the substring after a specified delimiter.
                    By.xpath("//input[substring-after(@name, '_') = 'name']");
                    
                    // normalize-space() - Removes leading and trailing spaces from a string.
                    By.xpath("//div[normalize-space(text())='Username']");
                    
                    // boolean() - Converts a value to a boolean.
                    By.xpath("//input[boolean(@disabled)]");
                    
                    // not() - Returns the negation of a boolean value.
                    By.xpath("//input[not(@type='hidden')]");
                    
                    // name() - Returns the name of a node.
                    By.xpath("//input[name()='username']");
                    
                    // local-name() - Returns the local name of a node (excluding the namespace prefix).
                    By.xpath("//*[local-name()='input']");
                    
                    // count() - Returns the number of nodes matching a given expression.
                    By.xpath("count(//input)");
                    
                    // sum() - Returns the sum of numeric values in a set of nodes.
                    By.xpath("//input[sum(//prices) > 100]");
                    
                    // position() - Returns the position of a node in a node set.
                    By.xpath("(//input[@type='text'])[position()=2]");
                    
                    // last() - Returns the last node in a node set.
                    By.xpath("(//input[@type='text'])[last()]");
