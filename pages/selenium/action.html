// click() - Clicks on a web element.
new Actions(driver).click(element).perform();
// clickAndHold() - Clicks (without releasing) at the current mouse location or on a given element.
new Actions(driver).clickAndHold(element).perform();
// contextClick() - Performs a right-click at the current mouse location or on a given element.
new Actions(driver).contextClick(element).perform();
// doubleClick() - Performs a double-click at the current mouse location or on a given element.
new Actions(driver).doubleClick(element).perform();
// dragAndDrop() - Drags an element from a source location and drops it at a target location.
new Actions(driver).dragAndDrop(source, target).perform();
// dragAndDropBy() - Drags an element from its current location and drops it at an offset.
new Actions(driver).dragAndDropBy(element, 50, 100).perform();
// moveByOffset() - Moves the mouse from the current position by the given offset.
new Actions(driver).moveByOffset(50, 100).perform();
// moveToElement() - Moves the mouse to the middle of the given element.
new Actions(driver).moveToElement(element).perform();
// release() - Releases the pressed mouse button at the current mouse location or on a given element.
new Actions(driver).release().perform();
// sendKeys() - Sends a sequence of keys to the currently focused element.
new Actions(driver).sendKeys(element, Keys.ENTER).perform();
// keyDown() - Performs a key press without releasing it.
new Actions(driver).keyDown(Keys.CONTROL).perform();
// keyUp() - Releases a pressed key.
new Actions(driver).keyUp(Keys.CONTROL).perform();
// build() - Generates a composite action containing all actions so far.
Action action = new Actions(driver).moveToElement(element).click().build();
action.perform();
// perform() - A convenience method for performing the actions built so far.
new Actions(driver).moveToElement(element).click().perform();
